db.createCollection("tipos_habitacion", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: [ "tipo" ],
      properties: {
        tipo: {
          bsonType: "string"
        }
      }
    }
  }
});

db.createCollection("habitaciones", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: [ "_id", "capacidad", "costo", "tipo", "comedor", "cocina", "jacuzzi" ],
      properties: {
        _id: {
          bsonType: "double"
        },
        capacidad: {
          bsonType: "double"
        },
        costo: {
          bsonType: "double"
        },
        consumos: {
          bsonType: "array"
        },
        tipo: {
          bsonType: "string"
        },
      }
    }
  }
});
